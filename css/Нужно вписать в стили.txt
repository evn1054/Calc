/* Очистим все отступы и границы */
*{
    padding: 0;
    margin: 0;
    border: 0;
}

/* Меняет подсчёт ширины блока для удобства верстки */
*,*:before, *:after {
    -moz-box-sizing: border-box;
    -webkit-box-sizing: border-box;
    box-sizing: border-box;
}

/* Убераем линии обвотки у некоторых браузеров */
:focus,:active{outline: none;}
a:focus,a:active{outline: none;}

/* Делаем блочными некоторые HTML теги */
nav,footer,header,aside{display: block;}

/* Уравняем поведение шрифтов, тегов для разных браузеров, в том числе запрещает самовольное увеличение шрифта на айфонах например */
html,body{
    height: 100%;
    width: 100%;
    font-size: 100%;
    line-height: 1;
    font-size: 14px;
    -ms-text-size-adjust: 100%;
    -moz-text-size-adjust: 100%;
    -webkit-text-size-adjust: 100%;
}

/* Просим наследовать семейстов шрифтов */
input,button,textarea{font-family: inherit;}

/* убирают некоторые особенности в некоторых браузерах */
input::-ms-clear{display: block;}
button{cursor: pointer;}
button::-moz-focus-inner{padding: 0;border: 0;}
a, a:visited{text-decoration: none;}
a:hover{text-decoration: none;}
ul li{list-style: none;}
img{vertical-align: top;}

/* Обнуляем заголовки  */
h1,h2,h3,h4,h5,h6{font-size: inherit; font-weight: 400;}
